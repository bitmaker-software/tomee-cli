016  (ns ^{:author "Daniel Cunha (soro) "}
017   tomee-cli.configuration.database-resource
018    (:require [tomee-cli.resources   :refer (add-resource xml-with-out-str define-resource) :as resource]
019              [tomee-cli.environment :refer (tomee-home) :as environment]))
020  
021  (defn define-datasource-resource
022    "Define a DataSource resource"
023    [id jdbc-drive jdbc-url username password jtamanaged]
024    (let [content (str "\njdbcDriver=" jdbc-drive "\njdbcUrl=" jdbc-url "\nusername=" username "\npassword=" password "\nJtaManaged=" jtamanaged "\n")]
025      (resource/define-resource id "javax.sql.DataSource" content)))
026  
027  (defn add-new-datasource-resource
028    "Write a new DataSource Resource in tomee.xml"
029    ([id jdbc-drive jdbc-url username password jtamanaged] (add-new-datasource-resource environment/tomee-home id jdbc-drive jdbc-url username password jtamanaged))
030    ([path id jdbc-drive jdbc-url username password jtamanaged] (resource/add-new-resource path (define-datasource-resource id jdbc-drive jdbc-url username password jtamanaged))))
