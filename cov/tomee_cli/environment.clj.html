018   tomee-cli.environment
019    (:require [clojure.java.shell :refer (sh)]
020              [clojure.string     :refer (split)]
021              [tomee-cli.utils    :refer (pretty-output)]
022              [environ.core       :refer (env)]))
023  
024  (def tomee-home (let [env-var (env :tomee-home)]
025                    (if (nil? env-var) "." env-var)))
026  
027  (defn tomee-xml-path
028    ([] (tomee-xml-path tomee-home))
029    ([path] (str path "/conf/tomee.xml")))
030  
031  (defn server-xml-path
032    ([] (server-xml-path tomee-home))
033    ([path] (str path "/conf/server.xml")))
034  
035  (defn windows?
036    "Verify if the operating system is windows"
037    []
038    (= (.toLowerCase (System/getProperty "os.name")) "windows"))
039  
040  (def extension (if (windows?) ".exe" ".sh"))
041  
042  (defn version
043    "Prints version numbers of the environment elements"
044    ([]     (version tomee-home))
045    ([path] (pretty-output (get (sh (str path "/bin/version" extension)) :out))))
