016  (ns ^{:author "Daniel Cunha (soro) ,
017                 Hildeberto Mendon√ßa "}
018   tomee-cli.execution
019    (:require [clojure.java.shell    :refer (sh)]
020              [tomee-cli.environment :refer (tomee-home extension) :as env]))
021  
022  (defn start
023    "Startup the Apache TomEE Server"
024    ([]     (start env/tomee-home))
025    ([tomee-path] (sh (str tomee-path "/bin/startup" env/extension))))
026  
027  (defn stop
028    "Shutdown the Apache TomEE Server"
029    ([]     (stop env/tomee-home))
030    ([tomee-path] (sh (str tomee-path "/bin/shutdown" env/extension))))
031  
032  (defn restart
033    "Shutdown and start the Apache TomEE Server"
034    ([]     (restart env/tomee-home))
035    ([tomee-path] (do (stop  tomee-path)
036                      (start tomee-path))))
