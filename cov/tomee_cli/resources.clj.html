016  (ns ^{:author "Daniel Cunha (soro) "}
017   tomee-cli.resources
018    (:require [clojure.xml           :refer (parse)   :as xml]
019              [clojure.java.io       :refer (as-file) :as io]
020              [tomee-cli.environment :refer (tomee-xml-path) :as environment]))
021  
022  (defn parse-xml
023    [path]
024    (xml/parse
025     (io/as-file path)))
026  
027  (defn xml-with-out-str
028    [xml-file]
029    (with-out-str
030      (xml/emit xml-file)))
031  
032  (defn add-resource
033    [path resource]
034    (let [xml-file (parse-xml path)
035          content  (xml-file :content)]
036      (cond
037        (nil? content) (assoc xml-file :content [resource])
038        :else (assoc xml-file :content (concat content [resource])))))
039  
040  (defn add-new-resource
041    ([path resource-defined]
042     (let [new-tomee-xml (add-resource (environment/tomee-xml-path path) resource-defined)
043           str-new-tomee-xml (xml-with-out-str new-tomee-xml)]
044       (spit (environment/tomee-xml-path path) str-new-tomee-xml)
045       str-new-tomee-xml)))
046  
047  (defn define-resource
048    [id resource-type content]
049    {:tag :Resource :attrs {:id (str id) :type (str resource-type)} :content [content]})
