020  (defn pretty-output [text]
021    "Formats a text to be beautifully printed by the repl."
022    (loop [out (split text #"\n")]
023      (if (empty? out)
024        "--------------"
025        (let [to-print (first out)]
026          (println to-print)
027          (recur (rest out))))))
028  
029  (defn filename-extension [filename]
030    (let [point-pos (.lastIndexOf filename ".")]
031      (if (< point-pos 0)
032        nil
033        (.toLowerCase (.substring filename (inc point-pos))))))
034  
035  (defn filename-from-path [path]
036    "Identifies and returns a file name present in a path."
037    (let [point-pos (.lastIndexOf path ".")
038          bar-pos   (.lastIndexOf path "/")]
039      (if (< point-pos 0)
040        nil
041        (.substring path (inc bar-pos)))))
